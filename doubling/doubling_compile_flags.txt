# doubling_compile_flags.txt
# Flags de compilación para el proyecto cobaya doubling
#
# CTRL_MIDI_PORT: puerto ALSA al que conecta midi_io al arrancar.
#   La cadena MIDI es: Oxygen 25 → H4MIDI WC → BLE → WIDI Core → Pi (snd-virmidi).
#   El Pi no ve "Oxygen 25" directamente; ve "Virtual Raw MIDI X-0".
#   Se deja vacío para que midi_io cree el puerto sin conectar —
#   el wrapper doubling_app conecta VirMIDI → pisound-control via aconnect.
#   Para ver puertos disponibles: aconnect -l
#
# CTRL_CONFIG_FILE: ruta relativa desde donde se ejecuta el binario
#   (doubling_app se ejecuta desde ~/pisound/doubling/)

-DPORTABLE_WORDSIZES
-DENABLE_CONTROL
-DENABLE_OSC
-DBLE_MIDI
-DCTRL_MIDI_PORT=""
-DCTRL_CONFIG_FILE="doubling_ctrl_config.txt"
-DCTRL_NUM_PARAMS=4
-DCTRL_SAMPLE_RATE=48000
-DCTRL_FRAME_SIZE=128
-DCTRL_OSC_TX_NUM=4
-DCTRL_TX_CONFIG_FILE="ctrl_tx.conf"
